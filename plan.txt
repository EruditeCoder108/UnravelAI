## STEP 4 — Mermaid Utilities in `App.jsx`

Add these utility functions near the top of `App.jsx`, before the component definition. These convert schema data to Mermaid syntax — the model never writes Mermaid directly.

```js
// ─── Mermaid Utilities ───────────────────────────────────────────────────────

// Sanitize any string into a valid Mermaid node ID
const mId = (s) => String(s).replace(/[^a-zA-Z0-9]/g, '_').slice(0, 30);

// Detect cycles in an edge list — returns true if cycle exists
function hasCycle(edges) {
    if (!edges || edges.length === 0) return false;
    const graph = {};
    edges.forEach(({ from, to }) => {
        if (!graph[from]) graph[from] = [];
        graph[from].push(to);
    });
    const visited = new Set();
    const check = (node, path) => {
        if (path.has(node)) return true;
        if (visited.has(node)) return false;
        visited.add(node); path.add(node);
        for (const next of (graph[node] || [])) {
            if (check(next, path)) return true;
        }
        path.delete(node);
        return false;
    };
    return edges.some(({ from }) => check(from, new Set()));
}

// Build sequence diagram from timelineEdges
function buildTimelineMermaid(edges) {
    if (!edges || edges.length === 0) return null;
    const lines = ['sequenceDiagram'];
    edges.forEach(({ from, to, label, isBugPoint }) => {
        const arrow = isBugPoint ? '-->>' : '->>';
        const prefix = isBugPoint ? '    Note over ' + mId(from) + ',' + mId(to) + ': 🐛 BUG HERE\n' : '';
        lines.push(`${prefix}    ${mId(from)}${arrow}${mId(to)}: ${label || ''}`);
    });
    return lines.join('\n');
}

// Build hypothesis elimination flowchart from hypothesisTree
function buildHypothesisMermaid(tree) {
    if (!tree || tree.length === 0) return null;
    const lines = ['flowchart TD'];
    tree.forEach(({ id, text, status, reason }) => {
        const nodeId = mId(id);
        const shortText = text.slice(0, 40);
        if (status === 'survived') {
            lines.push(`    ${nodeId}["${id}: ${shortText}"]`);
            lines.push(`    ${nodeId} --> SURVIVED["✅ Root Cause Confirmed"]`);
            lines.push(`    style ${nodeId} fill:#00ff88,color:#000`);
            lines.push(`    style SURVIVED fill:#00ff88,color:#000`);
        } else {
            const elimId = mId(id + '_elim');
            lines.push(`    ${nodeId}["${id}: ${shortText}"]`);
            lines.push(`    ${nodeId} -->|"${(reason || '').slice(0, 35)}"| ${elimId}["❌ Eliminated"]`);
            lines.push(`    style ${nodeId} fill:#333,color:#fff`);
            lines.push(`    style ${elimId} fill:#ff3333,color:#fff`);
        }
    });
    return lines.join('\n');
}

// Build AI loop cycle diagram from aiLoopEdges
function buildAILoopMermaid(edges) {
    if (!edges || edges.length === 0) return null;
    const lines = ['flowchart LR'];
    edges.forEach(({ from, to, label, isEscapePath }) => {
        const f = mId(from); const t = mId(to);
        lines.push(`    ${f}["${from}"] -->|"${label || ''}"| ${t}["${to}"]`);
        if (isEscapePath) {
            lines.push(`    style ${f} fill:#00ff88,color:#000`);
            lines.push(`    style ${t} fill:#00ff88,color:#000`);
        }
    });
    return lines.join('\n');
}

// Build variable mutation flow from variableStateEdges (only for complex variables)
function buildVariableMermaid(varEdges) {
    if (!varEdges || varEdges.length === 0) return null;
    const result = [];
    varEdges.forEach(({ variable, edges }) => {
        if (!edges || edges.length < 5) return; // only complex variables
        const lines = [`flowchart LR`];
        lines.push(`    DECL["${variable} declared"]`);
        edges.forEach(({ from, to, label, type }, i) => {
            const f = mId(from + i); const t = mId(to + i);
            lines.push(`    ${f}["${from}"] -->|"${label}"| ${t}["${to}"]`);
            if (type === 'write') lines.push(`    style ${f} fill:#ffaa00,color:#000`);
            if (type === 'read') lines.push(`    style ${t} fill:#448aff,color:#fff`);
            if (type === 'mutate') lines.push(`    style ${f} fill:#ff003c,color:#fff`);
        });
        result.push({ variable, mermaid: lines.join('\n') });
    });
    return result;
}

// Build data flow flowchart from flowchartEdges (Explain Mode)
function buildDataFlowMermaid(edges) {
    if (!edges || edges.length === 0) return null;
    if (hasCycle(edges)) return null; // fall back to table on cycles
    const lines = ['flowchart TD'];
    const seen = new Set();
    edges.forEach(({ from, to, label }) => {
        const f = mId(from); const t = mId(to);
        if (!seen.has(f)) { lines.push(`    ${f}["${from}"]`); seen.add(f); }
        if (!seen.has(t)) { lines.push(`    ${t}["${to}"]`); seen.add(t); }
        lines.push(`    ${f} -->|"${label || ''}"| ${t}`);
    });
    return lines.join('\n');
}

// Build dependency graph from dependencyEdges (Explain Mode)
function buildDependencyMermaid(deps) {
    if (!deps || deps.length === 0) return null;
    const lines = ['graph LR'];
    deps.forEach(({ file, imports }) => {
        (imports || []).forEach(imp => {
            lines.push(`    ${mId(file)}["${file}"] --> ${mId(imp)}["${imp}"]`);
        });
    });
    return lines.join('\n');
}

// Mermaid renderer component — renders the diagram or falls back to null
function MermaidChart({ chart, caption }) {
    const ref = React.useRef(null);
    React.useEffect(() => {
        if (!ref.current || !chart) return;
        ref.current.removeAttribute('data-processed');
        ref.current.innerHTML = chart;
        window.mermaid?.run({ nodes: [ref.current] });
    }, [chart]);
    if (!chart) return null;
    return (
        <div className="mermaid-wrap">
            <div className="mermaid" ref={ref}>{chart}</div>
            {caption && <p className="mermaid-caption">{caption}</p>}
        </div>
    );
}
```

```jsx
        </table>
    </section>
)}

{/* ── DEPENDENCY MAP + CHART ── */}
{safeExplain.componentMap.length > 0 && (
    <section className="report-section">
        <h2>Component & Dependency Map</h2>
        <MermaidChart
            chart={buildDependencyMermaid(safeExplain.dependencyEdges)}
            caption="File-level import dependencies — explicit imports only"
        />
        {safeExplain.componentMap.map((comp, i) => (
            <div key={i} className="component-block">
                <strong>{comp.name}</strong>
                {comp.children?.length > 0 && (
                    <div>Dependencies: {comp.children.join(', ')}</div>
                )}
                {comp.stateOwned?.length > 0 && (
                    <div>State: {comp.stateOwned.join(', ')}</div>
                )}
            </div>
        ))}
    </section>
)}

{/* ── KEY PATTERNS ── */}
{safeExplain.keyPatterns.length > 0 && (
    <section className="report-section">
        <h2>Key Patterns</h2>
        <ul>
            {safeExplain.keyPatterns.map((p, i) => <li key={i}>{p}</li>)}
        </ul>
    </section>
)}

{/* ── NON-OBVIOUS INSIGHTS ── */}
{safeExplain.nonObviousInsights.length > 0 && (
    <section className="report-section">
        <h2>Non-Obvious Insights</h2>
        <p className="section-subtitle">Things that would surprise a developer reading this for the first time</p>
        <ul>
            {safeExplain.nonObviousInsights.map((insight, i) => (
                <li key={i} className="insight-item">💡 {insight}</li>
            ))}
        </ul>
    </section>
)}

{/* ── GOTCHAS ── */}
{safeExplain.gotchas.length > 0 && (
    <section className="report-section">
        <h2>Gotchas</h2>
        <p className="section-subtitle">Hidden landmines — things that break when changed</p>
        {safeExplain.gotchas.map((g, i) => (
            <div key={i} className="gotcha-block">
                <div className="gotcha-title">⚠️ {g.title}</div>
                <p>{g.description}</p>
                {g.location && <code className="gotcha-location">📍 {g.location}</code>}
            </div>
        ))}
    </section>
)}

{/* ── ONBOARDING ── */}
{safeExplain.onboarding.length > 0 && (
    <section className="report-section">
        <h2>Onboarding Guide</h2>
        <p className="section-subtitle">Exactly where to go for the most common tasks</p>
        {safeExplain.onboarding.map((item, i) => (
            <div key={i} className="onboarding-block">
                <div className="onboarding-task">🎯 {item.task}</div>
                <div><strong>Where:</strong> <code>{item.whereToLook}</code></div>
                <div><strong>Model after:</strong> {item.patternToFollow}</div>
            </div>
        ))}
    </section>
)}

{/* ── ARCHITECTURE DECISIONS ── */}
{safeExplain.architectureDecisions.length > 0 && (
    <section className="report-section">
        <h2>Architecture Decisions</h2>
        {safeExplain.architectureDecisions.map((d, i) => (
            <div key={i} className="arch-block">
                <div className="arch-decision"><strong>{d.decision}</strong></div>
                {d.visibleReason && <div><strong>Why (visible in code):</strong> {d.visibleReason}</div>}
                {d.tradeoff && <div><strong>Tradeoff:</strong> {d.tradeoff}</div>}
            </div>
        ))}
    </section>
)}
```
